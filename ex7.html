Dois prédios e uma bolota amarela

Exemplo sobre o funcionamento de animação

A bolinha vai da esquerda para a direita
#include "playcb.h"
#include <stdio.h>
int main(){
Ponto p;
float t;
AbreJanela(500,500, "Uma cidade de dois predios");
PintarFundo(255, 255, 255);
MostraPlanoCartesiano(10);
printf("Vou desenhar uma cidade");
CriaGrupo(); //Este grupo é o grupo dos prédios
p.x = -100;
p.y = -20;
CriaRetangulo(30, 90, p);
Pintar(255, 0, 0);
p.x = -60;
p.y = -20;
CriaRetangulo(30, 50, p);
printf("\nVou desenhar uma bolinha que se mexe~~");
CriaGrupo(); //Este é o grupo da bolinha. Bolinha é independente dos prédios
p.x = -100;
p.y = -40;
CriaCirculo(10, p);
Pintar(255, 255, 0);
t = 0;
p.x = 0;
p.y = 0;
do{
p.x = t;
Move(p);
Desenha1Frame(); //Aqui eu renderizo apenas uma cena
t += 0.01;
}while(p.x < 210);
printf("\nA bolinha foi embora! D: Aperte ESC para sair");
Desenha(); //Aqui eu mando renderizar para sempre sem ficar mudando a cena
return 0;
}
