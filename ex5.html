Árvore recursiva
Uma árvore pequenininha feita com recursão
#include "playcb.h"
#include <time.h>
#define RAD 0.01745
void arvore(Ponto p1, float altura1, float angulo1, float profundidade){
Ponto p2;
float altura2, angulo2;
srand(time(NULL));
if(0 < profundidade){
p2.x = p1.x + altura1 * cos(angulo1 * RAD);
p2.y = p1.y + altura1 * sin(angulo1 * RAD);
CriaReta(p1, p2);
Pintar(0, 255, 0);
Grafite(2.0);
altura2 = altura1 * (float)(rand()%2);
angulo2 = angulo1 + (float)(20 + rand()%30);
arvore(p2, altura2, angulo2, profundidade - 1);
angulo2 = angulo1 - (float)(20 + rand()%30);
arvore(p2, altura2, angulo2, profundidade - 1);
}
}
int main(){
Ponto p;
p.x = 0;
p.y = 0;
AbreJanela(800,600, "Exemplo 5 - Arvore recursiva");
MostraPlanoCartesiano(10);
PintarFundo(255, 255, 255);
arvore(p, 5, 90, 5);
Desenha();
return 0;
}
